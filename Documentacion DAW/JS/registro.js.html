/**
 * AJAX Request Check if the nick is in use or not
 * @param txt
 */
function compruebaNick(txt){
    $.ajaxSetup({
        headers: {
            'X-CSRF-TOKEN': $('meta[name="csrf-token"]').attr('content')
        }
    });

    $.ajax({
        url : 'registro/alias',
        method: 'POST',
        dataType: 'JSON',
        data : {
            txt : txt
        },
    })
        .done(function (data) {
            if (data.repe) {
                $(".rep1").removeAttr("hidden");
                $("#env").prop("disabled","e")
            } else {
                $(".rep1").prop("hidden","e");
                $("#env").removeAttr("disabled")
            }

        })
}
